```c++
void linxing(int n) {
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			if (i > n / 2 + 1) {
				if ((j == -n / 2 + i) || (j == n + n / 2 - i + 1)) {
                    //这个是后半段每排应该打出“*”的下标与i的关系的判断式
					cout << "*";
				}
				else {
					cout << " ";
				}

					cout << " ";

			}
```
